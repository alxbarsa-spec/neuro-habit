        function openHelp(type) {
            const data = calibrationData[type];
            document.getElementById('modal-title').innerText = data.title;

            document.getElementById('text-ten').innerText = data.ten;

                        <h4 class="text-[10px] font-black text-slate-400 uppercase tracking-widest">Оценка по шкале:</h4>`;
            data.levels.forEach(l => {
                html += `<div class="flex gap-4 items-center">

                content.classList.replace('opacity-0', 'opacity-100');
                content.classList.replace('scale-95', 'scale-100');
            }, 10);
        }

        function closeHelp() {
            const modal = document.getElementById('help-modal');
            const content = document.getElementById('modal-content');
            content.classList.replace('opacity-100', 'opacity-0');
            content.classList.replace('scale-100', 'scale-95');
            setTimeout(() => modal.classList.add('hidden'), 300);
        }

        async function getAIAnalysis() {
            const loader = document.getElementById('loader');
            const resBox = document.getElementById('ai-text');
            loader.classList.remove('hidden');

            const dop = document.getElementById('i-dop').value;
            const oxy = document.getElementById('i-oxy').value;
            const ser = document.getElementById('i-ser').value;
            const end = document.getElementById('i-end').value;

            const systemPrompt = "Ты эксперт по методу Лоретты Бройнинг. Шкала 0-10 (0 - полный дефицит ресурса, 10 - макс ресурс). Проанализируй данные, укажи на критический гормон и дай 1 совет на 5 минут. Пиши на русском, коротко.";
            const userPrompt = `Ресурсы: Д:${dop}, О:${oxy}, С:${ser}, Э:${end}. Дай вердикт.`;

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: userPrompt }] }],
                        systemInstruction: { parts: [{ text: systemPrompt }] }
                    })
                });
                const data = await response.json();
                resBox.innerText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Ошибка.";
            } catch (e) { resBox.innerText = "Связь потеряна."; } 
            finally { loader.classList.add('hidden'); }
        }
